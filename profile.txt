        // MetaMask Wallet Connect
        async function connectWallet() {
    if (typeof window.ethereum !== 'undefined') {
        try {
            const accounts = await ethereum.request({ method: 'eth_requestAccounts' });
            const walletAddress = accounts[0];

            // Save wallet to server
            const res = await fetch('/profile/save-wallet', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ walletAddress })
            });

            const data = await res.json();
            if (data.success) {
                Swal.fire({
                    icon: 'success',
                    title: 'Wallet Connected!',
                    text: walletAddress,
                    timer: 1500,
                    showConfirmButton: false
                }).then(() => {
                    window.location.reload();
                });
            } else {
                Swal.fire("Error", "Failed to save wallet", "error");
            }

        } catch (err) {
            console.error("Connection error", err);
            Swal.fire("Error", "Wallet connection failed", "error");
        }
    } else {
        // ðŸ” Deep link to MetaMask mobile app (fallback)
        const isMobile = /Android|iPhone|iPad|iPod/i.test(navigator.userAgent);
        if (isMobile) {
            window.location.href = "https://metamask.app.link/dapp/http//192.168.137.1" ;
        } else {
            Swal.fire({
                title: "MetaMask Not Detected",
                text: "Please install MetaMask extension to continue.",
                icon: "warning",
                confirmButtonText: "Install MetaMask"
            }).then(() => {
                window.open("https://metamask.io/download/", "_blank");
            });
        }
    }
}

        // Token transfer functionality
        let selectedTokenId;
        let selectedTokenSymbol;

        function openTransferModal(tokenId, tokenName, tokenSymbol) {
            if (!window.ethereum || !window.ethereum.selectedAddress) {
                Swal.fire({
                    title: 'Wallet Required',
                    text: 'Please connect your MetaMask wallet before transferring tokens',
                    icon: 'warning',
                    confirmButtonText: 'Connect Wallet',
                    showCancelButton: true
                }).then((result) => {
                    if (result.isConfirmed) {
                        connectWallet();
                    }
                });
                return;
            }

            selectedTokenId = tokenId;
            selectedTokenSymbol = tokenSymbol;
            document.getElementById('transferTokenId').value = tokenId;
            
            // Update modal title with token info
            document.querySelector('#transferModal .modal-title').innerHTML = 
                `Transfer ${tokenSymbol} Tokens <small class="text-muted">${tokenName}</small>`;

            const modal = new bootstrap.Modal(document.getElementById('transferModal'));
            modal.show();
        }

        document.getElementById('transferForm').addEventListener('submit', async function(e) {
            e.preventDefault();

            const recipient = document.getElementById('recipientAddress').value;
            const amount = document.getElementById('transferAmount').value;

            if (!ethers.isAddress(recipient)) {
                Swal.fire('Invalid Address', 'Please enter a valid Ethereum address', 'error');
                return;
            }

            if (!window.ethereum || !window.ethereum.isMetaMask) {
                Swal.fire('MetaMask Required', 'You need MetaMask to send tokens', 'error');
                return;
            }

            try {
                Swal.fire({
                    title: 'Processing Transaction',
                    html: 'Please confirm the transaction in your MetaMask wallet...',
                    allowOutsideClick: false,
                    didOpen: () => {
                        Swal.showLoading();
                    }
                });

                const provider = new ethers.BrowserProvider(window.ethereum);
                const signer = await provider.getSigner();

                const contract = new ethers.Contract(
                    '<%= contractAddress %>',
                    <%- JSON.stringify(erc721ABI) %>,
                    signer
                );

                const tx = await contract.safeTransferFrom(
                    signer.address,
                    recipient,
                    selectedTokenId,
                    amount,
                    '0x'
                );

                Swal.fire({
                    title: 'Transaction Submitted!',
                    html: `Your transfer of ${amount} ${selectedTokenSymbol} tokens has been initiated.<br>
                           <a href="https://sepolia.etherscan.io/tx/${tx.hash}" target="_blank">View on Etherscan</a>`,
                    icon: 'success',
                    confirmButtonText: 'OK'
                }).then(() => {
                    const modal = bootstrap.Modal.getInstance(document.getElementById('transferModal'));
                    modal.hide();
                });

            } catch (err) {
                console.error(err);
                let errorMsg = 'Transfer failed';
                if (err.message.includes('user rejected transaction')) {
                    errorMsg = 'Transaction was rejected';
                } else if (err.message.includes('insufficient balance')) {
                    errorMsg = 'Insufficient token balance';
                }
                Swal.fire('Error', errorMsg, 'error');
            }
        });

            card.addEventListener('animationend', () => {
                card.classList.remove('animate-pop');
            });
        });

function copyToClipboard(elementId) {
  const element = document.getElementById(elementId);
  if (element) {
    navigator.clipboard.writeText(element.textContent.trim())
      .then(() => {
        // Show a small toast or tooltip to indicate success
        const originalText = element.textContent;
        element.textContent = "Copied!";
        setTimeout(() => {
          element.textContent = originalText;
        }, 2000);
      })
      .catch(err => {
        console.error("Failed to copy:", err);
      });
  }
}
